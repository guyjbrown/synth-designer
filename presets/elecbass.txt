# parameters

@param cutoff
type : float
mutable: yes
step : 0.01
min : 100
max : 12000
default : 12000
doc : cutoff frequency in Hz
@end

@param decay
type : float
mutable: yes
step : 0.01
min : 0.1
max : 12
default : 2
doc : filter decay time in sec
@end

@param detune
type : float
mutable: yes
step : 0.01
min : 0
max : 10
default : 3.3
doc : detune in cents
@end

@param volume
type : float
mutable: yes
step : 0.01
min : 0
max : 1
default : 0.8
doc : master volume
@end

CUSTOM-OSC : osc2 TABLE basspick

# patch

osc1.out -> mix.in
osc2.out -> mix.in

lowpass.out -> vca.in
vca.out -> audio.in

# tweak

osc1.pitch = param.pitch
osc2.pitch = param.pitch
osc1.detune = param.detune
osc2.detune = -1*param.detune
vca.level = param.volume*param.level
env.level = 500+(param.cutoff-500)*param.level
env.decay = param.decay
env.attack = 0
env.out -> lowpass.cutoffCV
lowpass.resonance = -10
mix.level = 0.5
