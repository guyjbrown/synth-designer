@end

# parameters

@param cutoff
type : float
mutable: yes
step : 0.01

default : 1000

@end

VCA : filtermod

# patch

osc.out -> lowpass.in
lowpass.out -> vca.in
vca.out -> audio.in

lfo1.out -> pitchmod.in
pitchmod.out -> osc.pitchCV
lfo1.pitch = 3.8
pitchmod.level = 1.4

lfo2.out -> filtermod.in
filtermod.out -> lowpass.cutoffCV
lfo2.pitch = 5
filtermod.level = param.cutoff*0.2

env.out -> vca.levelCV

# tweak

env.level = param.level
env.attack = 0.1
env.release = 0.1
env.decay = 0.5
env.sustain = param.level*0.9

osc.pitch = param.pitch
lowpass.cutoff = param.cutoff

# end
